\begin{abstract}
This document contains the solution to find a unit vector that makes an angle of 90\degree,135\degree,45\degree with positive X,Y and Z axis respectively.
\end{abstract}

 \textbf{Soln} : Given,
 
 \begin{align}
 \vec m = \myvec{\cos 90\degree     \\ \cos 135\degree\\ \cos45\degree}
 \label{eq1}
 \end{align}

we know that,
\begin{equation}
  \vec m=\frac{\vec m }{\norm{\vec m }}
\end{equation}
Also,
\begin{equation}
{\norm{\vec m }}= \sqrt{0^2+\brak{\frac{-1}{\sqrt{2}}}^2+(\frac{1}{\sqrt{2}})^2} 
\implies{\norm{\vec m }}=1
\label{eq2}
\end{equation}

 Hence,From \eqref{eq1} and \eqref{eq2} we have the unit vector:
 \begin{align}
\vec m  = \myvec{ 0 \\ \frac{-1} {\sqrt{2}}\\ \frac{1}{\sqrt{2}}}
  \end{align}
